#

#   reportlab    — pure-Python PDF rendering (WeasyPrint dropped because

#                  button; reviewers can edit + sign + save-as-PDF themselves
#   markdown     — Markdown→HTML conversion that feeds both render_pdf
#                  (ReportLab Paragraph accepts a small HTML subset) and
#                  the Streamlit "Rendered Document" tab
reportlab>=4.0,<5.0
python-docx>=1.1,<2.0
markdown>=3.7,<4.0
